fh_make_castle_button = {
    scope = province
	saved_scopes = {
		builder
    }
    is_shown = {
		NOR = {
			has_holding_type = castle_holding
			has_holding_type = cow_metropolis_holding 
		}
		fh_can_convert_holding_trigger = yes
    }
    is_valid = {
        scope:builder = {
			gold >= 300
		}
    }
	ai_is_valid = {
		always = no
	}
    effect = {
        set_variable = fh_convert_to_castle
		custom_tooltip = fh_convert_to_castle.tt
		fh_start_convert_holding_type_effect = yes
        province_owner = {
        	remove_long_term_gold = 300
        }
    }
}
